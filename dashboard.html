1. Open your Google Sheet
2. Extensions → Apps Script
3. Delete all existing code, paste the code below → Save (Ctrl+S)
4. Deploy → New Deployment → Web App
5. Execute as: Me  |  Who has access: Anyone
6. Click Deploy → Copy the Web App URL → paste above
function doPost(e) { try { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheet = ss.getSheetByName("Leads") || ss.getActiveSheet(); if (sheet.getLastRow() === 0) { sheet.appendRow([ "App Name","Developer","Email","Category", "Installs","Score","Description","URL", "Icon","Keyword","Scraped At","Email Sent" ]); } var r = data.row; sheet.appendRow([ r.app_name, r.developer, r.email, r.category, r.installs, r.score, r.description, r.url, r.icon, r.keyword, r.scraped_at, r.email_sent ]); return ContentService .createTextOutput(JSON.stringify({status:"ok"})) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({status:"error",msg:err.toString()})) .setMimeType(ContentService.MimeType.JSON); } }
